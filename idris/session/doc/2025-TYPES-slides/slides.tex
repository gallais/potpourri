    \; ::= \; \recvar{x}
    \; | \; \smallest{x}{p}
    \; | \; \largest{x}{p} \\
    &
    \; | \; \send{A}{p}
    \; | \; \recv{A}{p}

    \; | \; \offer{p}{q}
