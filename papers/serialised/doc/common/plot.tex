\newcommand{\graphOf}[3]{\begin{tikzpicture}
  \begin{axis}
    [ xlabel = depth
    , ylabel = #3 (ns)
    , ymode = log
    , legend pos = north west
    ]
    \addplot table
      [ grid = major
      , x = size
      , y = #1
      , col sep = comma
      ] {#3.csv};
    \addplot table
      [ grid = major
      , x = size
      , y = #2
      , col sep = comma
      ] {#3.csv};
  \legend{#1, #2}
  \end{axis}
\end{tikzpicture}}
